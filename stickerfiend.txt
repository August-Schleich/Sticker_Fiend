NAME='stickerfiend_django''
DJANGODIR=/webapps/stickerfiend/stickerfiend_django
SOCKFILE=/webapps/stickerfiend/environment_3_8_2/run/gunicorn.sock
USER=stickerfienduser
GROUP=webapps
NUM_WORKERS=3
DJANGO_SETTINGS_MODULE=stickerfiend_django.settingsprod
DJANGO_WSGI_MODULE=stickerfiends_django.wsgi
TIMEOUT=120

cd $DJANGODIR
source ../bin/activate
export DJANGO_SETTINGS_MODULE=$DJANGO_SETTINGS_MODULE
export PYTHONPATH=$DJANGODIR:$PYTHONPATH

RUNDIR=$(dirname $SOCKFILE)
test -d $RUNDIR || mkdir -p $RUNDIR

exec ../environment_3_8_2/bin/gunicorn ${DJANGO_WSGI_MODULE}:application \
--name $NAME \
--workers $NUM_WORKERS \
--timeout $TIMEOUT \
--user=$USER --group=$GROUP \
--bind=unix:$SOCKFILE \
--log-level=debug \
--log-file=-
